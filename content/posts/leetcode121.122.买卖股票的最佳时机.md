int maxProfit121(vector<int>& prices) {
    //è´ªå¿ƒ
    if (!prices.size())
        return 0;
    int tmpM = prices[0], res = 0;
    for (auto x : prices) {
        if (x < tmpM) {
            tmpM = x;
        } else {
            res = max(res, x-tmpM);
        }
    }
    return res;
}

int maxProfit122(vector<int>& prices) {
    int res = 0;
    for (int i = 1; i < prices.size(); i++) {
