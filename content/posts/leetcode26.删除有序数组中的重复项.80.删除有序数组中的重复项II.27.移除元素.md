/*
 * 27
 * in: [0,1,2,2,3,0,4,2], val = 2
 * out: 5, nums = [0,1,4,0,3]
 */
int removeElement(vector<int>& nums, int val) {
    int i = 0;
    for (int j = 0; j < nums.size(); ++j) {
        if (nums[j] != val) {
            nums[i] = nums[j];
            i++;
        }
    }
    return i;
}
